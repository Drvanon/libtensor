#
#	Set up libtensor as an external library
#

include(${LIBVMM_DIR}/cmake/DepsLibTensor.txt)

#	Debugging options
#
add_definitions(-DLIBTENSOR_DEBUG)

#	Include and library paths
#
include_directories(${LIBTENSOR_DIR})
add_library(tensor STATIC IMPORTED)
set_target_properties(tensor PROPERTIES
	IMPORTED_LOCATION ${LIBVMM_DIR}/build/libtensor.a
)
set(LIBTENSOR_LIBS libtensor ${LIBTENSOR_DEP_LIBS})
